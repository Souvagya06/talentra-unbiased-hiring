🟢 STEP 2: Reset Function
<script>
function resetPassword() {
  const email = document.getElementById("email").value;

  firebase.auth().sendPasswordResetEmail(email)
    .then(function () {
      alert("Reset email sent!");
    })
    .catch(function (error) {
      alert(error.message);
    });
}
</script>
